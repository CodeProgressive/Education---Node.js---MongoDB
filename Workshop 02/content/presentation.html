$ node app [Name]

The answer


// Require some node modules
var colors = require("colors"),
    moment = require("moment");

// If no name is inserted, stop the node and show an error
if(typeof process.argv[2] == 'undefined') {
    console.log("Please insert your name".red);
    // Exit using 1 means there was an error
    process.exit(1);
}

// Create the strings to show
var greet = "Hi " + process.argv[2],
    time = "it’s " + moment().format('MMMM Do YYYY, h:mm:ss a') + " right now!";

// Now show them while adding the color
console.log(greet.green + ", " + time.yellow);

Doing it safely, with protection

Keep your code healthy, you can't trust anyone

var dope = (function() {

    var self = {};

    // This is publicly available, accessible by dope.shit()
    self.shit = function() {
        return safeShit();
    }

    // This is private and object only, accessible by dope.shit()
    var safeShit = function() {
        return "This is really safe, no one can mess with your shit!";
    }

    return self;

})();

// Exporting public parts only
module.exports = dope;
